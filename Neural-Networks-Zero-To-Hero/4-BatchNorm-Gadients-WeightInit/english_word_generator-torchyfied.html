def encode_word(word, prev_tokens):
    def _encode(wrd):
        return [stoi[w] for w in wrd]
    prefix_word = "." * prev_tokens + word
    suffix_word = word + "."
    return list(windowed(_encode(prefix_word), prev_tokens)), _encode(suffix_word)

def encode_words(words, prev_tokens):
    X = []
    Y = []
    for word in words:
        x, y = encode_word(word, prev_tokens)
        X.extend(x)
        Y.extend(y)

    X = torch.tensor(X)
    Y = torch.tensor(Y)
    print(X.shape, Y.shape)
    return X, Y

class Embedding:
    def __init__(
        self,
        vocab_size: int,
        embedding_size: int,
    ) -> None:
        self.embs = torch.randn(size=(vocab_size, embedding_size), generator=GEN)

        self._out: torch.Tensor

    def __call__(self, x: torch.Tensor) -> torch.Tensor:
        self._out = self.embs[x]
        return self.out

    @property
    def parameters(self) -> list[torch.Tensor]:
        return [self.embs]

    @property
    def out(self) -> torch.Tensor:
        return self._out

class Linear:
    def __init__(self,
        fan_in: int,
        fan_out: int,
        bias: bool = True,
    ) -> None:
        self.weights = torch.randn(size=(fan_in, fan_out), generator=GEN) / fan_in ** 0.5
        self.bias = torch.zeros(fan_out) if bias is True else None

        self._out: torch.Tensor

    def __call__(self, x: torch.Tensor) -> torch.Tensor:
        self._out = x @ self.weights
        if self.bias is not None:
            self._out += self.bias
        return self.out

    @property
    def parameters(self) -> list[torch.Tensor]:
        return [self.weights] + ([] if self.bias is None else [self.bias])

    @property
    def out(self) -> torch.Tensor:
        return self._out

class BatchNorm1d:
    def __init__(
        self,
        fan_in: int,
        eps: float = 1e-5,
        momentum: float = 0.001,
        training: bool = True,
    ) -> None:
        self.gamma = torch.ones(fan_in)
        self.beta = torch.zeros(fan_in)

        self.running_mean = torch.zeros(fan_in, requires_grad=False)
        self.running_var = torch.ones(fan_in, requires_grad=False)
        self.momentum = momentum

        self.eps = eps
        self.training = training

        self._out: torch.Tensor

    def __call__(self, x: torch.Tensor) -> torch.Tensor:
        x_mean = x.mean(dim=0, keepdim=True) if self.training else self.running_mean
        x_var = x.var(dim=0, keepdim=True) if self.training else self.running_var
        x_hat = (x - x_mean) / (x_var + self.eps).sqrt()
        self._out = self.gamma * x_hat + self.beta
        if self.training:
            with torch.no_grad():
                self.running_mean = (1 - self.momentum) * self.running_mean + self.momentum * x_mean
                self.running_var = (1 - self.momentum) * self.running_var + self.momentum * x_var
        return self.out

    @property
    def parameters(self) -> list[torch.Tensor]:
        return [self.gamma, self.beta]

    @property
    def out(self) -> torch.Tensor:
        return self._out
