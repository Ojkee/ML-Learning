from torch.nn import Module, Embedding
import torch.nn.functional as F
from fastai.layers import sigmoid_range

class CollaborativeFilteringModel(Module):
    def __init__(self, n_users, n_books, n_factors):
        super().__init__()
        self.user_embeddings = Embedding(n_users, n_factors)
        self.user_bias = Embedding(n_users, 1)
        
        self.book_embeddings = Embedding(n_books, n_factors)
        self.book_bias = Embedding(n_books, 1)

    def forward(self, x): 
        users = x[:, 0]
        books = x[:, 1]
        y_users = self.user_embeddings(users) 
        y_books = self.book_embeddings(books)
        y = (
            (y_users * y_books).sum(dim=1, keepdim=True)
            + self.user_bias(users)
            + self.book_bias(books)
        )
        return sigmoid_range(y, 1, 10.5)
