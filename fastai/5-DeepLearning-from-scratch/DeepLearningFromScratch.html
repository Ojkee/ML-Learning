def init_random_vector(n: int):
    return torch.rand(n) - 0.5

def calc_predictions(coeffs, indeps):
    return (indeps * coeffs).sum(axis=1)

def calc_loss(coeffs, indeps, deps):
    return torch.abs(calc_predictions(coeffs, indeps) - deps).mean()

from itertools import pairwise

def init_coeffs(hidden_dims: list[int]):
    sizes = [INPUT_DIM] + hidden_dims + [1]
    n_layers = len(sizes)

    def make_layer(m, n):
        return torch.rand(m, n) - 0.3

    def make_const():
        return (torch.rand(1)[0] - 0.5) * 0.1

    layers = [make_layer(*p) for p in pairwise(sizes)]
    consts = [make_const() for _ in range(n_layers - 1)]

    for layer in layers + consts:
        layer.requires_grad_()

    return layers, consts

from torch import sigmoid, relu

def calc_predictions(coeffs, indeps):
    layers, consts = coeffs
    n_layers = len(layers)
    res = indeps

    def is_last_layer(i): return i == n_layers - 1

    for i, (layer, const) in enumerate(zip(layers, consts)):
        res = res @ layer + const
        if not is_last_layer(i):
            res = relu(res)
    return sigmoid(res)

def update_coeffs(coeffs, learning_rate):
    layers, consts = coeffs
    for layer in layers + consts:
        layer.sub_(layer.grad * learning_rate)
        layer.grad.zero_()

def execute_epoch(coeffs, learning_rate, indeps, deps):
    loss = calc_loss(coeffs, indeps, deps)
    loss.backward()
    with torch.no_grad():
        update_coeffs(coeffs, learning_rate)

def make_and_train_model(
    hidden_dims: list[int],
    learning_rate: float,
    epochs: int,
    indeps,
    deps,
):
    coeffs = init_coeffs(hidden_dims)
    for i in range(epochs):
        execute_epoch(coeffs, learning_rate, indeps, deps)
        if (i < 100 and i % 10 == 0) or (i >= 100 and i % 100 == 0):
            print(f"Iter: {i}, loss: {calc_loss(coeffs, indeps, deps).item()}")
    return coeffs

def accuracy(coeffs, indeps, deps):
    preds = calc_predictions(coeffs, indeps)
    right_preds = deps.bool() == (preds > 0.5)
    return right_preds.float().mean()
