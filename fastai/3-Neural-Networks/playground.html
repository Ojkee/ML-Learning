def mk_f(a11, b11, a12, b12, a21, b21):
    f11 = mk_lin(a11, b11)
    f12 = mk_lin(a12, b12)
    r1 = lambda x: torch.relu(f11(x))
    r2 = lambda x: torch.relu(f12(x))
    f21 = mk_lin(a21, b21)
    return lambda x: f21(r1(x) + r2(x))
